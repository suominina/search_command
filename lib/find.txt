find [オプション] [path] [式] //指定したディレクトリ以下で条件に合致するフィルを検索
    (式)-name //指定したファイル名で検索
	-type //ファイルタイプで検索
		find . -type l //カレントディレクトリ以下のシンボリックリンク
	-print //検索結果を標準出力
	-exec command ¥; //検索後、コマンドを実行
	-atime //指定した日時を基に、最終アクセスがあったファイルを検索
		find / -atime -7 // /ディレクトリ以下で直近一週間にアクセスしたファイル
	-mtime //指定した日時をもとに、最終更新されたファイルを検索
		find / -mtime +7 // /ディレクトリ以下で一週間以上前に編集したファイル
	-user //ファイルの所有者ユーザ名またはUIDを指定
		find /bin -uid 0 -perm /4000 // /binディレクトリ内でファイルの所有者がroot(0)かつSUIDビットが立っているファイル(-perm /4000)
	-maxdepth //検索する深さ(階層)を指定
